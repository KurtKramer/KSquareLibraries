364 void ScannerFile4BitEncoded::BuildConversionTables ()
365 {
366  GlobalGoalKeeper::StartBlock ();
367  if (!convTable4BitTo8Bit)
368  {
369  kkint32 x = 0;
370  kkint32 y = 0;
371  kkint32 inc = 256 / 16;
372  convTable4BitTo8Bit = new uchar[16];
373  convTable8BitTo4Bit = new uchar[256];
374  for (x = 0; x < 16; ++x)
375  {
376  kkint32 this8Bit = x * inc;
377  kkint32 next8Bit = (x + 1) * inc;
378 
379  kkint32 fourBitTo8BitNum = (kkint32)(this8Bit + (kkint32)(((float)x / 16.0f) * (float)inc));
380 
381  convTable4BitTo8Bit[x] = (uchar)fourBitTo8BitNum;
382  for (y = this8Bit; y < next8Bit; ++y)
383  convTable8BitTo4Bit[y] = x;
384  }
385 
386  compensationTable = new uchar[256];
387  for (kkint16 pv = 0; pv < 256; ++pv)
388  {

390  compensationTable[pv] = convTable4BitTo8Bit[encodedValue];

393  }

567  kkuint32 lineBuffSize

569 {
570  bool eol = false;
571  uchar opCode = 0;
572  OpRec rec;
573  OpRec rec2;
574  kkuint32 recsRead = 0;
575 
576  kkuint32 bufferLineLen = 0;
577 
578  do
579  {
580  recsRead = fread (&rec, sizeof (rec), 1, file);
581  if (recsRead == 0)
582  {
583  break;
584  }
585 
586  opCode = rec.textBlock1.opCode;
587 
588  if (opCode == 0)
589  eol = true;
590 
591  else if (bufferLineLen >= lineBuffSize)
592  {
593  // Something has gone wrong, we should have encountered a eol opCode before this point; that is the
594  // length of the encoded line is exceeding the scan line length.
595  eol = true;
596  ungetc (rec.textChar, file);
597  break;
598  }
599 
600 
601  else if (opCode == 1)
602  ProcessTextBlock (rec);
603 
604  else if (opCode == 2)
605  ProcessInstrumentDataWord (rec);
606 
607  else if ((opCode >= 4) && (opCode <= 9))
608  {
609  // OpCode determines RunLen (4=2, 5=3, ... 9=7)
610  kkuint32 runLen = opCode - 2;
611  runLenChar = convTable4BitTo8Bit [rec.runLen.pixelValue];
612  kkuint32 newLineSize = bufferLineLen + runLen;
613 
614  if (newLineSize > lineBuffSize)
615  {
616  cerr << "ScannerFile4BitEncoded::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << nextScanLine << "]." << endl;
617  cerr << " newLineSize: " << newLineSize << endl;
618  }
619  else
620  {
621  memset (&(lineBuff[bufferLineLen]), runLenChar, runLen);
622  bufferLineLen = newLineSize;
623  }
624  }
625 
626  else if (opCode == 10) /* OpRecRun256Len1 */
627  {
628  recsRead = fread (&rec2, sizeof (rec2), 1, file);
629  if (recsRead < 1)
630  eol = true;
631  else
632  {
633  kkuint32 runLen = 1 + rec2.run256Len2.runLen;
634  kkuint32 newLineSize = bufferLineLen + runLen;
635  if (newLineSize > lineBuffSize)
636  {
637  cerr << "ScannerFile4BitEncoded::GetNextScanLine ***ERROR*** Exceeding 'bufferLineLen'; ScanLine[" << nextScanLine << "]." << endl;
638  }
639  else
640  {
641  uchar runLenChar = convTable4BitTo8Bit [rec.run256Len1.pixelValue];
642  memset (&(lineBuff[bufferLineLen]), runLenChar, runLen);
643  bufferLineLen = newLineSize;
644  }
645  }
646  }
647 
648  else if (opCode == 11)
649  {
650  if (bufferLineLen < lineBuffSize)
651  {
652  lineBuff[bufferLineLen] = convTable4BitTo8Bit[rec.raw1Pixel.pixelValue];
653  ++bufferLineLen;
654  }
655  else

658  }
659  }
660

664  kkuint16 numRawPixels = numRawRecs * 2;
665  kkuint16 newBufferLineLen = bufferLineLen + numRawPixels;
666  if (newBufferLineLen > lineBuffSize)
667  {

669  }
670  ProcessRawPixelRecs (numRawRecs, lineBuff, lineBuffSize, bufferLineLen);
671  }
672 
673  else if (opCode == 13)
674  {
675  recsRead = fread (&rec2, sizeof (rec2), 1, file);
676  if (recsRead < 1)
677  {
678  eol = true;
679  eof = true;
680  }
681  else
682  {

684  kkuint16 numRawPixels = 1 + 2 * numRawRecs;
685  kkuint16 newBufferLineLen = bufferLineLen + numRawPixels;
686  if (newBufferLineLen > lineBuffSize)

689  }

694  ++bufferLineLen;

699  }

868 void ScannerFile4BitEncoded::WriteNextScanLine (const uchar* buffer,
869  kkuint32 bufferLen
870  )
871 {
872  encodedBuffNext = encodedBuff;
873  encodedBuffLen = 0;

875  if (bufferLen > (encodedBuffLeft - 10))
876  {
877  kkuint32 newEncodedBuffSise = bufferLen + 10;

879  }
880 
881  curCompStatus = csNull;
882 
883  for (kkuint16 x = 0; x < bufferLen; ++x)
884  {
885  uchar nextCh = convTable8BitTo4Bit[buffer[x]];
886 
887  switch (curCompStatus)
888  {
889  case csNull:
890  {
891  runLen = 1;
892  runLenChar = nextCh;
893  curCompStatus = csRunLen;
894  }
895  break;
896 
897  case csRunLen:
898  {
899  if (nextCh == runLenChar)
900  {
901  ++runLen;
902  }
903  else if (runLen == 1)
904  {
905  rawStr[0] = runLenChar;
906  rawStr[1] = nextCh;
907  rawStrLen = 2;
908  runLen = 0;
909  curCompStatus = csRaw;
910  }
911  else
912  {
913  AddCurRunLenToOutputBuffer ();
914  runLenChar = nextCh;
915  runLen = 1;
916  }
917  }
918  break;
919 
920  case csRaw:
921  {
922  if (rawStrLen == 0)
923  {
924  rawStr[rawStrLen] = nextCh;
925  ++rawStrLen;
926  }
927 
928  else if (rawStr[rawStrLen - 1] != nextCh)
929  {
930  rawStr[rawStrLen] = nextCh;
931  ++rawStrLen;
932  }
933 
934  else
935  {
936  --rawStrLen;
937  if (rawStrLen > 0)
938  AddCurRawStrToOutputBuffer ();
939 
940  runLen = 2;
941  runLenChar = nextCh;
942  curCompStatus = csRunLen;
943  }
944  }
945  break;
946  }
947  }
948 
949  if (curCompStatus == csRunLen)
950  AddCurRunLenToOutputBuffer ();
951 
952  if (curCompStatus == csRaw)
953  AddCurRawStrToOutputBuffer ();
954 
955  OpRec rec;
956  rec.endOfScanLine.opCode = 0;
957  rec.endOfScanLine.filler = 0;
958  *encodedBuffNext = rec;
959  ++encodedBuffNext;
960 
961  encodedBuffLen = encodedBuffNext - encodedBuff;
962  fwrite (encodedBuff, sizeof (OpRec), encodedBuffLen, file);
963  fileSizeInBytes = osFTELL (file);
964 
965  /**@todo Write out OutputBuffer; */
966 
967 } /* WriteNextScanLine */
