195  kkint32 MemoryConsumedEstimated () const
196  {
197  kkint32 memoryConsumedEstimated = sizeof (svm_model)
198  + param.MemoryConsumedEstimated ()
199  + exampleNames.size () * 40;
200 
201  if (SV) memoryConsumedEstimated += sizeof (svm_node*) * l;
202  if (sv_coef) memoryConsumedEstimated += (nr_class - 1) * sizeof (double*) + l * (nr_class - 1) * sizeof (double);
203  if (rho) memoryConsumedEstimated += l * sizeof (double);
204  if (label) memoryConsumedEstimated += nr_class * sizeof (kkint32);
205  if (nSV) memoryConsumedEstimated += nr_class * sizeof (kkint32);
206  if (featureWeight) memoryConsumedEstimated += dim * sizeof (double);
207  if (kValueTable) memoryConsumedEstimated += sizeof (double) * l;
208  if ((xSpace != NULL) && weOwnXspace)
209  memoryConsumedEstimated += sizeof (svm_node) * l;
210 
211  return memoryConsumedEstimated;
212  }
