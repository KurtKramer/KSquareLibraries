241  kkint32 srcColStart = Max (0, srcCol);
242  kkint32 srcColEnd = Min (srcCol + multiple, srcWidth);
243 
244  kkint32 count = 0;
245 
246  for (r = srcRowStart; r <srcRowEnd; ++r)
247  {
248  for (c = srcColStart; c <srcColEnd; ++c)
249  {
250  total += srcMatrix[r][c];
251  ++count;
252  }
