199  template<typename DftType>
200  void KK_DFT<DftType>::BuildMask ()
201  {
202  DftComplexType N((DftType)size, 0);
203  DftComplexType M((DftType)size, 0);
204 
205  kkint32 x;
206 
207  DftComplexType direction;
208  if (forwardTransform)
209  direction = DftComplexType ((DftType)-1.0, (DftType)0.0);
210  else
211  direction = DftComplexType ((DftType)1.0, (DftType)0.0);
212 
213  DftComplexType** fourierMask = new DftComplexType*[size];
214  for (x = 0; x < size; x++)
215  fourierMask[x] = new DftComplexType[size];
216 
217  DftComplexType j;
218  j = sqrt (MinusOne);
219 
220  for (kkint32 m = 0; m < size; m++)
221  {
222  DftComplexType mc ((DftType)m, (DftType)0);
223 
224  for (kkint32 k = 0; k < size; k++)
225  {
226  DftComplexType kc ((DftType)k, (DftType)0);
227  fourierMask[m][k] = exp (direction * j * Two * Pi * kc * mc / M);
228 
229  DftType exponetPart = (DftType)2.0 * (DftType)3.14159265359 * (DftType)k * (DftType)m / (DftType)size;
230  DftType realPart = cos (exponetPart);
231  DftType imgPart = -sin (exponetPart);
232 
233  if (realPart != fourierMask[m][k].real ())
234  {
235  continue;
236  }
237 
238  if (imgPart != fourierMask[m][k].imag ())
239  {

241  }

243  }
