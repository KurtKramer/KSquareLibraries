944 void Model::ReduceTrainExamples ()
945 {
946  kkint32 examplesPerClass = param->ExamplesPerClass ();
947  kkuint32 zed = 0;
948 
949  if (examplesPerClass < 0)
950  examplesPerClass = int32_max;
951 
952  bool reductionNeeded = false;
953 
954  {
955  // First lets see if reduction is even nessasary.
956  ClassStatisticListPtr stats = trainExamples->GetClassStatistics ();
957  if (!stats)
958  {
959  log.Level (-1) << endl
960  << "Model::ReduceTrainExamples ***ERROR*** can not retrieve Class Stat's for training data." << endl
961  << endl;
962  validModel = false;
963  return;
964  }
965 
966  for (zed = 0; (zed < stats->size ()) && (!reductionNeeded); zed++)
967  {
968  if (stats->IdxToPtr (zed)->Count () > (kkuint32)examplesPerClass)
969  reductionNeeded = true;
970  }
971 
972  delete stats;
973  }
974 
975  if (!reductionNeeded)
976  {
977  log.Level (20) << "Model::ReduceTrainExamples Was not needed. No classes exceeded 'examplesPerClass'." << endl;
978  return;
979  }
980 
981  FeatureVectorListPtr reducedSet = new FeatureVectorList (fileDesc, false, log);
982  FeatureVectorListPtr deleteSet = new FeatureVectorList (fileDesc, false, log); // Examples taht we do not use will need to be deleted.
983  MLClassList::iterator idx;
984 
985  for (idx = classes->begin (); idx != classes->end (); idx++)
986  {
987  MLClassPtr ic = *idx;
988  FeatureVectorListPtr examplesThisClass = trainExamples->ExtractImagesForAGivenClass (ic);
989  if ((!examplesThisClass) || (examplesThisClass->size () < 1))

993  << endl;
994  continue;
995  }
996

998  {
999  reducedSet->AddQueue (*examplesThisClass);
1000  }
1001  else
1002  {
1003  examplesThisClass->RandomizeOrder ();
1004  zed = 0;
1005  while (zed < (kkuint32)examplesPerClass)
1006  {
1007  reducedSet->PushOnBack (examplesThisClass->IdxToPtr (zed));
1008  zed++;
1009  }
1010  while (zed < examplesThisClass->size ())
1011  {
1012  deleteSet->PushOnBack (examplesThisClass->IdxToPtr (zed));
1013  zed++;
1014  }
1015  }
1016

1018  examplesThisClass = NULL;
1019  }
1020 
1021  if (weOwnTrainExamples)
1022  {
1023  trainExamples->Owner (false);
1024  delete trainExamples;
1025  trainExamples = reducedSet;
1026  reducedSet = NULL;
1027 
1028  trainExamples->Owner (true);
1029  deleteSet->Owner (true);
1030  delete deleteSet;
1031  deleteSet = NULL;
1032  }
1033  else
1034  {
1035  trainExamples = reducedSet;
1036  reducedSet = NULL;
1037  trainExamples->Owner (false);
1038  deleteSet->Owner (false);
1039  delete deleteSet;
1040  deleteSet = NULL;
1041  }
